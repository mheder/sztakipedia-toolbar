 54 /**
 55  * Uses the native parsing function of the browser to create a valid XML Document.
 56  * 
 57  * @note Taken from jQuery.parseXML() (see http://api.jquery.com/jQuery.parseXML/) We need this because the jQuery
 58  *       version used in MediaWiki/WikiEditor does not provide it (probably <1.5).
 59  * 
 60  * @param data {string} the XML as a string
 61  * @return Object the XML DOM, or <code>undefined</code> if <code>data</code> is <code>undefined</code>
 62  */
 63 SztakipediaClient.parseXML = function(data) {
 64 	if (typeof data == 'undefined')
 65 		return undefined;
 66 	var xml, tmp;
 67 
 68 	//alert('Raw: ' + data);
 69 
 70 	// FIXME hack to remove <!DOCTYPE ...> header
 71 	data = data.replace(/^<!DOCTYPE[^>]+>/, '');
 72 	// FIXME hack to close HTML-style elements in XHTML
 73 	data = data.replace(/(<(meta|link|img) [^>]+)>/i, '$1/>');
 74 
 75 	if (window.DOMParser) { // Standard
 76 		tmp = new DOMParser();
 77 		xml = tmp.parseFromString(data, "text/xml");
 78 	}
 79 	else { // IE
 80 		xml = new ActiveXObject("Microsoft.XMLDOM");
 81 		xml.async = "false";
 82 		xml.loadXML(data);
 83 	}
 84 
 85 	tmp = xml.documentElement;
 86 
 87 	if (!tmp || !tmp.nodeName || tmp.nodeName === "parsererror") {
 88 		//		alert("Invalid XML string: " + JSON.stringify(data, null, 2));
 89 		if (SztakipediaClient.getOption('debug'))
 90 			alert("Invalid XML string: '" + data + "'");
 91 		throw "XML not well-formed.";
 92 	}
 93 	//alert('XML: ' + JSON.stringify(xml, null, 2));
 94 	return xml;
 95 };

 97 /**
 98  * Read an XML elements attribute list into a hash.
 99  * @param {Node} elem The DOM element.
100  * @return Object
101  */
102 SztakipediaClient.xmlAttributesAsHash = function(elem) {
103 	if (typeof elem === 'undefined')
104 		return undefined;
105 	var item = {};
106 	for ( var i = 0; i < elem.attributes.length; i++) {
107 		var attrib = elem.attributes[i];
108 		item[attrib.name] = attrib.value;
109 	}
110 	return item;
111 };
112 
113 /**
114  * Read an HTML elements attribute list into a hash.
115  * @param {Node} elem The DOM element.
116  * @return Object
117  */
118 SztakipediaClient.htmlAttributesAsHash = function(elem) {
119 	if (typeof elem === 'undefined')
120 		return undefined;
121 	var item = {};
122 	for ( var i = 0; i < elem.attributes.length; i++) {
123 		var attrib = elem.attributes[i];
124 		if (attrib.specified == true) { // Works in most browsers
125 			item[attrib.name] = attrib.value;
126 		}
127 	}
128 	return item;
129 };

131 /**
132  * Get the string representation of a DOM XML node.
133  * @see adapted from http://forum.jquery.com/topic/jquery-serializing-xml#14737000000169757
134  * @param {Node} elem The DOM element.
135  */
136 SztakipediaClient.serializeXML = function(elem) {
137     var out;
138     if (typeof XMLSerializer === 'function') { // modern browsers
139         var xs = new XMLSerializer();
140         out = xs.serializeToString(elem);
141     } else if (elem && elem.xml !== 'undefined') { // IE
142         out = elem.xml;
143     }
144     return out;
145 };

147 /**
148  * Concatenate text nodes inside the element.
149  * @param {Node} elem The DOM element.
150  * @return string
151  */
152 SztakipediaClient.extractTextRecursive = function(elem) {
153 	if (elem == null)
154 		return "";
155 
156 	var acc = "";
157 	if (elem.nodeType == Node.TEXT_NODE) {
158 		acc += elem.nodeValue;
159 	}
160 	if (elem.hasChildNodes()) {
161 		var children = elem.childNodes;
162 		for ( var i = 0; i < children.length; i++) {
163 			acc += SztakipediaClient.extractTextRecursive(children[i]);
164 		}
165 	}
166 	return acc;
167 };
168 
169 /**
170  * Concatenate text nodes inside the element, up to immediate child elements.
171  * @param {Node} elem The DOM element. 
172  * @return string
173  */
174 SztakipediaClient.extractTextNonRecursive = function(elem) {
175 	if (elem == null) {
176 		return "";
177 	}
178 	else {
179 		var acc = "";
180 		if (elem.nodeType == Node.TEXT_NODE) {
181 			acc += elem.nodeValue;
182 		}
183 		if (elem.hasChildNodes()) {
184 			var children = elem.childNodes;
185 			for ( var i = 0; i < children.length; i++) {
186 				if (children[i].nodeType == Node.TEXT_NODE) {
187 					acc += children[i].nodeValue;
188 				}
189 			}
190 		}
191 		return acc;
192 	}
193 };

207 /**
208  * Assemble a query URL from a set of parameters.
209  * @param {Object} params A key-value map of the HTTP parameters. 
210  * @private
211  * @return string The URL to query the remote REST API.
212  */
213 SztakipediaClient.queryUrl = function(params) {
214 	if (typeof SztakipediaClient.getOption('json proxy url') === 'undefined')
215 		params['format'] = 'json';
216 	var url = SztakipediaClient.getOption('api url') + '?' + $j.param(params);
217 	if (typeof SztakipediaClient.getOption('json proxy url') !== 'undefined') {
218 		url = SztakipediaClient.getOption('json proxy url') + '?' + $j.param( {
219 			'url' : url
220 		});
221 	}
222 	//alert('URL: ' + url);
223 	return url;
224 };

226 /**
227  * Perform a GET request to the API endpoint.
228  * 
229  * For JSONP requests, the API response always returns a HTTP 200 OK status, while indicating actual status code (and probably error message) in the payload as follows.
230  * For successful requests:
231  * <blockquote>
232  * <code>json123(['...content...', 200])</code>
233  * </blockquote>
234  * For unsuccessful requests:
235  * <blockquote>
236  * <code>json123([null, 5xx, '...error...'])</code>
237  * </blockquote>
238  * @param {Object} params The HTTP request parameters to be passed.
239  * @param {function((string|undefined))} [callback] The function to be called with the HTTP response content {@link String}, or undefined on failure.
240  * @private
241  */
242 SztakipediaClient.doGet = function(params, callback) {
243 	var url = SztakipediaClient.queryUrl(params);
244 	$j.ajax( {
245 		'url' : url,
246 		'dataType' : 'jsonp',
247 		'jsonp' : 'callback',
248 		'success' : function(data) {
249 
250 			//alert(JSON.stringify(data, null, 2));
251 
252 		
253 			//alert("RAW: " + data);
254 			var content = data[0], 
255 				status = data[1], 
256 				error = (data.length > 2 ? data[2] : undefined);
257 			
258 			if (!(typeof error == 'undefined' || error == null)) {
259 				if (SztakipediaClient.getOption('debug'))
260 					alert('HTTP error ' + status + ' while retrieving "' + url + '": ' + JSON.stringify(data, null, 2));
261 				
262 				throw 'HTTP error ' + status + ' while retrieving "' + url + '"';
263 			}
264 			callback(content);
265 		}
266 	});
267 };

309 /**
310  * Retrieve the list of dialogs from the response DOM element.
311  * @param {Object} xml The XML DOM element to be processed.
312  * @return Object List of dialogs.
313  * @private
314  */
315 SztakipediaClient.parseDialog = function(xml) {
316 	if (typeof xml === 'undefined' || xml === null)
317 		throw "Undefined";
318 	
319 	var list = {};
320 	var builder = xml.documentElement.getAttribute('builder');
321 	var elems = xml.getElementsByTagName('Dialog');
322 	for ( var j = 0; j < elems.length; j++) {
323 		var elem = elems[j];
324 		var dialog = SztakipediaClient.xmlAttributesAsHash(elem);
325 		dialog['builder'] = builder;
326 		dialog['text'] = SztakipediaClient.extractTextNonRecursive(elem.getElementsByTagName('Text')[0]);
327 		dialog['suggestions'] = [];
328 
329 		var suggestionelems = elem.getElementsByTagName('Suggestion');
330 		for ( var k = 0; k < suggestionelems.length; k++) {
331 			var suggestionelem = suggestionelems[k];
332 
333 			var suggestion = SztakipediaClient.xmlAttributesAsHash(suggestionelem);
334 			suggestion['text'] = SztakipediaClient.extractTextNonRecursive(suggestionelem.getElementsByTagName('Text')[0]);
335 			if (suggestionelem.getElementsByTagName('SuggestionURL').length) {
336 				var url = SztakipediaClient.extractTextNonRecursive(suggestionelem.getElementsByTagName('SuggestionURL')[0]);
337 				if (url)
338 					suggestion['url'] = url;
339 			}
340 			if (suggestionelem.getElementsByTagName('Image').length) {
341 				suggestion['image'] = SztakipediaClient.xmlAttributesAsHash(suggestionelem.getElementsByTagName('FullResolution')[0]);
342 				suggestion['thumbnail'] = SztakipediaClient.xmlAttributesAsHash(suggestionelem.getElementsByTagName('Thumbnail')[0]);
343 			}
344 
345 			// extract content in different formats
346 			var wikitext = SztakipediaClient.extractTextNonRecursive(suggestionelem.getElementsByTagName('SuggestionContents')[0]
347 					.getElementsByTagName('WikitextFormat')[0]);
348 			if (wikitext)
349 				suggestion['content'] = wikitext;
350 
351 			var xhtml = SztakipediaClient.extractTextNonRecursive(suggestionelem.getElementsByTagName('SuggestionContents')[0]
352 					.getElementsByTagName('XhtmlFormat')[0]);
353 			if (xhtml)
354 				suggestion['html'] = xhtml;
355 
356 			// parse insertion strategies
357 			suggestion['insertionstrategies'] = {};
358 			var iselems = suggestionelem.getElementsByTagName('InsertionStrategies')[0].childNodes;
359 			for ( var l = 0; l < iselems.length; l++) {
360 				var iselem = iselems[l];
361 				switch (iselem.tagName) {
362 					case 'ContextSensitive':
363 						suggestion['insertionstrategies']['contextsensitive'] = {
364 							'replace' : SztakipediaClient.extractTextNonRecursive(iselem.getElementsByTagName('ContentToReplace')[0]),
365 							'before' : SztakipediaClient.extractTextNonRecursive(iselem.getElementsByTagName('ContentBefore')[0]),
366 							'after' : SztakipediaClient.extractTextNonRecursive(iselem.getElementsByTagName('ContentAfter')[0])
367 						};
368 						break;
369 					case 'CharacterPositionBased':
370 						// TODO convert strings (e.g. {"begin" : "4"}) to integer ({"begin" : 4}) 
371 						suggestion['insertionstrategies']['characterpositionbased'] = SztakipediaClient.xmlAttributesAsHash(iselem);
372 						break;
373 					case 'AbsolutePosition':
374 						suggestion['insertionstrategies']['absolute'] = SztakipediaClient.xmlAttributesAsHash(iselem);
375 						break;
376 					default:
377 						if (iselem.nodeType === Node.ELEMENT_NODE) {				
378 				 			if (SztakipediaClient.getOption('debug')) {
379 				 				alert("Unrecognized insertion policy: " + iselem.tagName + "\n" + SztakipediaClient.serializeXML(iselem));
380 								throw "Unrecognized insertion policy: " + iselem.tagName;
381 				 			}
382 						}
383 						break;
384 				}
385 			}
386 			dialog['suggestions'][suggestion['id']] = suggestion;
387 		}
388 
389 		list[dialog['id']] = dialog;
390 	}
391 	//alert('parseDialog: ' + JSON.stringify(list, null, 2));
392 	return list;
393 };

397  * @param {Object} xml The XML DOM element to be processed.
398  * @return string The session ID.
399  * @private
400  * 
401  */

404 		return undefined;

614 /**

617 SztakipediaClient.init = function() {
618 	if (SztakipediaClient.getOption('debug')) {
619 		// Provides cross-browser JSON serialization. For debugging purposes.
