122 	 * Get an option - user settings override site-wide which override defaults.
123 	 * @param {string|undefined} opt The name of the option.
124 	 * @return {string} The value of the option.
125 	 */
126 	SztakipediaTB.getOption = function(opt) {
127 		if (SztakipediaTB.UserOptions[opt] != undefined) {
128 			return SztakipediaTB.UserOptions[opt];
129 		}
130 		else if (SztakipediaTB.Options[opt] != undefined) {
131 			return SztakipediaTB.Options[opt];
132 		}
133 		return SztakipediaTB.DefaultOptions[opt];
134 	};

145 	/**
146 	 * Create a WikiEditor dialog from a {@link sztakipediaTemplate} object.
147 	 * @param {Object} tem The {@link sztakipediaTemplate} object to create a dialog for.
148 	 * @return {Object} A WikiEditor-style dialog descriptor.  
149 	 */
150 	SztakipediaTB.toDialog = function(tem) {
151 		var sform = SztakipediaTB.escStr(tem.shortform);
152 		var dialogid = 'sztakipedia-dialog-' + sform;
153 		var dialog = {
154 			resizeme : false,
155 			titleMsg : dialogid,
156 			id : 'sztakipediatoolbar-' + sform,
157 			init : function() {
158 			},
159 			html : tem.getInitial(),
160 			dialog : {
161 				width : 675,
162 				open : function() {
163 					$j(this).html(SztakipediaTB.getOpenTemplate().getForm());
164 					$j('.sztakipedia-prev-parse').bind('click', SztakipediaTB.prevParseClick);
165 				},
166 				beforeclose : function() {
167 					SztakipediaTB.resetForm();
168 				},
169 				buttons : {
170 					'sztakipedia-form-submit' : function() {
171 						$j.wikiEditor.modules.toolbar.fn.doAction($j(this).data('context'), {
172 							type : 'encapsulate',
173 							options : {
174 								peri : ' '
175 							}
176 						}, $j(this));
177 						var ref = SztakipediaTB.getRef(false, true);
178 						$j(this).dialog('close');
179 						$j.wikiEditor.modules.toolbar.fn.doAction($j(this).data('context'), {
180 							type : 'encapsulate',
181 							options : {
182 								pre : ref
183 							}
184 						}, $j(this));
185 					},
186 					'sztakipedia-form-showhide' : SztakipediaTB.showHideExtra,
187 					'wikieditor-toolbar-tool-link-cancel' : function() {
188 						$j(this).dialog('close');
189 					},
190 					'sztakipedia-form-reset' : function() {
191 						SztakipediaTB.resetForm();
192 					}
193 				}
194 			}
195 		};
196 		
197 		if (tem.preview)
198 			dialog['dialog']['buttons']['sztakipedia-refpreview'] = function() {
199 				var template = SztakipediaTB.getOpenTemplate();
200 				var ref = SztakipediaTB.getRef(!template.refs, false);
201 				var div = $j("#sztakipediatoolbar-" + sform);
202 				div.find('.sztakipedia-preview-label').show();
203 				div.find('.sztakipedia-ref-preview').text(ref).show();
204 				
205 				var container = div.find('.sztakipedia-form-container');
206 				var	scrollTo = div.find('.sztakipedia-ref-preview');
207 
208 	
209 				if (SztakipediaTB.getOption('autoparse')) {
210 					SztakipediaTB.scrollTo(scrollTo, container);
211 					SztakipediaTB.prevParseClick();
212 				}
213 				else {
214 					SztakipediaTB.scrollTo(scrollTo, container, 'fast');
215 					
216 					div.find('.sztakipedia-prev-parse').show();
217 					div.find('.sztakipedia-prev-parsed-label').hide();
218 					div.find('.sztakipedia-preview-parsed').html('');
219 				}
220 			};
221 		return dialog;
222 	};

793 		
794 		// TODO on forinsert == true, cross-call CiteTB
795 		var template = SztakipediaTB.getOpenTemplate();
796 		var templatename = template.templatename;

829 			}
830 		}
831 		if ($j('#sztakipedia-form-status').val() != 'closed') {
832 			for ( var i = 0; i < template.extra.length; i++) {
