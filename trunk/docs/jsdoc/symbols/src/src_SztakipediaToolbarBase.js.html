 26 // XML entity encode/decode by DracoBlue
 27 //
 28 // source: http://dracoblue.net/dev/encodedecode-special-xml-characters-in-javascript/155/
 29 //
 30 var xmlValue2EntityMap = {
 31 	'&' : '&',
 32 	'"' : '"',
 33 	'<' : '<',
 34 	'>' : '>'
 35 };
 36 function encodeXml(string) {
 37 	return string.replace(/([\&"<>])/g, function(str, item) {
 38 		return xmlValue2EntityMap[item];
 39 	});
 40 };
 41 var xmlEntity2ValueMap = {
 42 		'&' : '&',
 43 		'"' : '"',
 44 		'<' : '<',
 45 		'>' : '>'
 46 	};
 47 
 48 function decodeXml(string) {
 49 	return string.replace(/("|<|>|&)/g, function(str, item) {
 50 		return xmlEntity2ValueMap[item];
 51 	});
 52 }
 53 
 54 function decodeXmlJ(str) {
 55 	return $j('<div />').html(str.replace(/</g,'<')).text();
 56 }

 64 // Add a .non-empty class to 
 65 function adjustCssClassEmptiness() {
 66 	var el = $j(this);
 67 	if (el.val())
 68 	{
 69 		el.removeClass('empty');
 70 		el.addClass('non-empty');
 71 	}
 72 	else
 73 	{
 74 		el.removeClass('non-empty');
 75 		el.addClass('empty');
 76 	}

 79 /**
 80  * @class Object for cite templates. Stores the information needed to construct template filling dialog windows.
 81  * @constructor Creates a template object, given its name and available fields.
 82  * @param {string} templatename
 83  * @param {string} shortform
 84  * @param {Object} basicfields Basic fields of the template.
 85  * @param {Object} [expandedfields] Extra fields of the template. Fields supplied here shall be optional and not present in basic fields.
 86  * @private
 87  */
 88 function sztakipediaTemplate(templatename, shortform, basicfields, expandedfields) {
 89 	//alert("New template: " + templatename);
 90 	
 91 	/* Properties */
 92 	/**
 93 	 * The template name - "cite web", "cite book", etc.
 94 	 * @type string
 95 	 */
 96 	this.templatename = templatename;
 97 	/**
 98 	 * A short form, used for the dropdown box.
 99 	 * @type string
100 	 */
101 	this.shortform = shortform;
102 	/**
103 	 * Basic fields (e.g., author, title, publisher...)
104 	 * @type Object
105 	 */
106 	this.basic = basicfields;
107 	
108 	/**
109 	 * Less common - quote, archiveurl - should be everything the template supports minus the basic ones
110 	 * @type Object
111 	 */
112 	this.extra = expandedfields || null;
113 	
114 	/**
115 	 * Whether the template is a citation template.
116 	 * @type boolean
117 	 */
118 	this.refs = true;
119 	
120 	/**
121 	 * Whether previews are allowed for this template.
122 	 * Can be set to {@code false} for templates that do not generate output (e.g, PERSONDATA), or would generate too large HTML.
123 	 * @type boolean
124 	 */
125 	this.preview = true;
126 
127 	// Add it to the list
128 	SztakipediaTB.Templates[this.templatename] = this;
129 	
130 	// Methods
131 	/**
132 	 * Create a HTML fragment for a form with the given fields as inputs.
133 	 * @param {Object} fields The fields to be used for creating input widgets.
134 	 * @return {Object} The jQuery DOM element containing the HTML fragment.
135 	 */
136 	this.makeFormInner = function(fields) {
137 		var i = 0;
138 		var trs = new Array();
139 		for (i = 0; i < fields.length; i++) {
140 			var fieldobj = fields[i];
141 			var field = fieldobj.field;
142 			var ad = false;
143 			var im;
144 			if ($j.inArray(field, SztakipediaTB.getOption('autodate fields')) != -1) {
145 				im = $j('<img />')
146 					.attr('src', 'http://upload.wikimedia.org/wikipedia/commons/thumb/7/7b/Nuvola_apps_date.svg/20px-Nuvola_apps_date.svg.png')
147 					.attr('alt', mw.usability.getMsg('sztakipedia-insert-date'))
148 					.attr('title', mw.usability.getMsg('sztakipedia-insert-date'));
149 				ad = $j('<a/>')
150 					.attr('href', '#')
151 					.append(im)
152 					.attr('id', 'sztakipedia-date-' + SztakipediaTB.escStr(this.shortform) + '-' + field);
153 				$j('#sztakipedia-date-' + SztakipediaTB.escStr(this.shortform) + '-' + field).live('click', SztakipediaTB.fillAccessdate);
154 			}
155 
156 			if (fieldobj.autofillid) {
157 				var autotype = fieldobj.autofillid;
158 				im = $j('<img />')
159 					.attr('src', 'http://upload.wikimedia.org/wikipedia/commons/thumb/1/17/System-search.svg/18px-System-search.svg.png')
160 					.attr('alt', mw.usability.getMsg('sztakipedia-autofill-alt')).attr('title', mw.usability.getMsg('sztakipedia-autofill-alt'));
161 				ad = $j('<a />')
162 					.attr('href', '#')
163 					.append(im)
164 					.attr('id', 'sztakipedia-auto-' + SztakipediaTB.escStr(this.shortform) + '-' + field + '-' + autotype);
165 				$j('#sztakipedia-auto-' + SztakipediaTB.escStr(this.shortform) + '-' + field + '-' + autotype).live('click', SztakipediaTB.initAutofill);
166 			}
167 
168 			var display = mw.usability.getMsg('sztakipedia-' + field + '-label');
169 			// FIXME if (typeof display == 'undefined' || display == '' || display == false)
170 			display = SztakipediaTB.fixStr(field);
171 			var tooltip = fieldobj.tooltip ? $j('<abbr />').attr('title', mw.usability.getMsg(fieldobj.tooltip) || decodeXmlJ(fieldobj.tooltip)).text('*').addClass('sztakipedia-field-tooltip') : false;
172 
173 			var input = '';
174 			if (ad) {
175 				input = $j('<input tabindex="1" style="width:82%" type="text" /> ');
176 			} else {
177 				input = $j('<input tabindex="1" style="width:100%" type="text" />');
178 			}
179 			input.attr('id', 'sztakipedia-' + SztakipediaTB.escStr(this.shortform) + '-' + field);
180 			if (fieldobj.autofillprop) {
181 				input.addClass('sztakipedia-' + SztakipediaTB.escStr(this.shortform) + '-' + fieldobj.autofillprop);
182 			}
183 			input.change(adjustCssClassEmptiness).change();
184 			var label = $j('<label />');
185 			label.attr('for', 'sztakipedia-' + SztakipediaTB.escStr(this.shortform) + '-' + field).text(display);
186 			label.attr('title', 'raw field name: ' + field);
187 			if (tooltip) {
188 				label.append(tooltip);
189 			}
190 			var style = 'text-align:right; width:20%;';
191 			if (i % 2 == 1) {
192 				style += ' padding-left:1em;';
193 			} else {
194 				var tr = $j('<tr />');
195 			}
196 			var td1 = $j('<td class="sztakipedia-form-td" />').attr('style', style);
197 			td1.append(label);
198 			tr.append(td1);
199 			var td2 = $j('<td class="sztakipedia-form-td" style="width:30%" />');
200 			td2.append(input);
201 			if (ad) {
202 				td2.append(ad);
203 			}
204 			tr.append(td2);
205 			if (i % 2 == 0) {
206 				trs.push(tr);
207 			}
208 		}
209 		return trs;
210 
211 	};
212 
213 	/**
214 	 * Gives a little bit of HTML so the open form can be identified. 
215 	 * Also required to interoperate with wikieditor, which checks for {@code $j(...).length == 0} to determine emptiness.
216 	 * @return {Object} a jQuery DOM element
217 	 */
218 	this.getInitial = function() {
219 		var hidden = $j('<input type="hidden" class="sztakipedia-template" />');
220 		hidden.val(this.templatename);
221 		return hidden;
222 	};
223 
224 	/**
225 	 * Makes the form used in the dialog boxes.
226 	 * @return {Object} a jQuery DOM element
227 	 */
228 	this.getForm = function() {
229 		var main = $j("<div class='sztakipedia-form-container' />");
230 		var form1 = $j('<table style="width:90%; background-color:transparent;" class="sztakipedia-basic-fields" />');
231 		var i;
232 		var trs = this.makeFormInner(this.basic);
233 		for ( i = 0; i < trs.length; i++) {
234 			form1.append(trs[i]);
235 		}
236 
237 		if (this.extra) {
238 			var form2 = $j('<table style="width:100%; background-color:transparent; display:none" class="sztakipedia-extra-fields">');
239 			trs = this.makeFormInner(this.extra);
240 			for ( i = 0; i < trs.length; i++) {
241 				form2.append(trs[i]);
242 			}
243 		  main.append(form1).append(form2);
244 		}
245 
246 		if (this.refs) {
247 			var form3 = $j('<table style="width:95%; background-color:transparent;padding-top:1em" class="sztakipedia-other-fields">');
248 			var tr = $j('<tr />');
249 			var td1 = $j('<td class="sztakipedia-form-td" style="text-align:right; width:20%" />');
250 			var label1 = $j('<label />');
251 			label1.attr('for', 'sztakipedia-' + SztakipediaTB.escStr(this.shortform) + '-name').text(mw.usability.getMsg('sztakipedia-name-label'));
252 			td1.append(label1);
253 			var td2 = $j('<td class="sztakipedia-form-td" style="width:30%" />');
254 			var input1 = $j('<input tabindex="1" style="width:auto" type="text" />');
255 			input1.attr('id', 'sztakipedia-' + SztakipediaTB.escStr(this.shortform) + '-name');
256 			td2.append(input1);
257 			var td3 = $j('<td class="sztakipedia-form-td" style="text-align:right; padding-left:1em; width:20%">');
258 			var label2 = $j('<label />');
259 			label2.attr('for', 'sztakipedia-' + SztakipediaTB.escStr(this.shortform) + '-group').text(mw.usability.getMsg('sztakipedia-group-label'));
260 			td3.append(label2);
261 			var td4 = $j('<td class="sztakipedia-form-td" style="width:30%" />');
262 			var input2 = $j('<input tabindex="1" style="width:auto" type="text" />');
263 			input2.attr('id', 'sztakipedia-' + SztakipediaTB.escStr(this.shortform) + '-group');
264 			td4.append(input2);
265 			tr.append(td1).append(td2).append(td3).append(td4);
266 			form3.append(tr);
267 			main.append(form3);
268 		}
269 		var extras = $j('<div />');
270 		extras.append('<input type="hidden" class="sztakipedia-form-status" value="closed" />');
271 		var hidden = $j('<input type="hidden" class="sztakipedia-template" />');
272 		hidden.val(this.templatename);
273 		extras.append(hidden);
274 		
275 		if (this.preview) {
276 			var span1 = $j('<span class="sztakipedia-preview-label" style="display:none;" />');
277 			span1.text(mw.usability.getMsg('sztakipedia-raw-preview'));
278 			extras.append(span1).append('<div class="sztakipedia-ref-preview" style="padding:0.5em; font-size:110%" />');
279 			var span2 = $j('<span class="sztakipedia-prev-parsed-label" style="display:none;" />');
280 			span2.text(mw.usability.getMsg('sztakipedia-parsed-label'));
281 			extras.append(span2).append('<div class="sztakipedia-preview-parsed" style="padding-bottom:0.5em; font-size:110%" />');
282 			var link = $j('<a href="#" class="sztakipedia-prev-parse" style="margin:0 1em 0 1em; display:none; color:darkblue" />');
283 			link.text(mw.usability.getMsg('sztakipedia-form-parse'));
284 			extras.append(link);
285 		}
286 		
287 		main.append(extras);
288 
289 		return main;
290 	};
291 }
